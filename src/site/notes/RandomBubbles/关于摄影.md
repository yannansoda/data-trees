const tag = "#摄影";
const folder = "RandomBubbles/FlashThoughts";

// 获取所有 Markdown 文件
const files = app.vault.getMarkdownFiles().filter(f => f.path.startsWith(folder));

for (let file of files) {
  const content = await app.vault.read(file);
  // 逐行找出以 #标签 开头的句子或段落
  const lines = content.split("\n").filter(line =>
    line.trim().startsWith(tag) || line.includes(`${tag} `)
  );

  if (lines.length > 0) {
    dv.header(3, dv.fileLink(file.path));
    for (let line of lines) {
      // 清理前后空格，输出整句话
      dv.paragraph("> " + line.trim());
    }
  }
}
